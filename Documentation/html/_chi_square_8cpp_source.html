59size_t group_sample(long long* sample, double* expected_probabilities, size_t outcome_n, size_t sample_size)
60{
61 size_t outcomes_in_sample{outcome_n};
62 for (size_t i{}; i < outcome_n; ++i)
63 {
64 while (sample_size * expected_probabilities[i] < 5 && i < outcome_n - 1)
65 {
66 sample[i + 1] += sample[i];
67 sample[i] = -1;
68 expected_probabilities[i + 1] += expected_probabilities[i];
69 expected_probabilities[i++] = 0;
70 --outcomes_in_sample;
71 }
72 if (i == outcome_n - 1 && sample_size * expected_probabilities[i] < 5)
73 {
74 size_t j{i - 1};
75 for (; sample[j] == -1; --j);
76 sample[i] += sample[j];
77 sample[j] = -1;
78 expected_probabilities[i] += expected_probabilities[j];
79 expected_probabilities[j] = 0;
80 --outcomes_in_sample;
81 break;
82 }
83 }
84 return outcomes_in_sample;
85}

87double criterion(const long long* sample, size_t sample_size, const double* expected_probabilities, size_t outcome_n)
88{
89 double crit{};
90 for (unsigned int i{}; i < outcome_n; ++i)
91 {
92 if (sample[i] != -1)
93 crit += (sample[i] - sample_size * expected_probabilities[i]) * (sample[i] - sample_size *
94 expected_probabilities[i]) /
95 (sample_size * expected_probabilities[i]);
96 }
97 return crit;
98}

100void ChiSquare::CalculateStatistic()
101{
102 if (has_generator)
103 _observed->Simulate();
104
105 size_t states = _expected.GetStatesNumber();
106 size_t sample_size = _observed->GetSampleSize();
107
108 auto expp_copy = new double[states];
109 auto sample_copy = new long long[states];
110 memcpy(expp_copy, _expected.GetProbabilities(), states * sizeof(double));
111 memcpy(sample_copy, _observed->GetSample(), states * sizeof(long long));
112
113 size_t outcomes_in_sample = group_sample(sample_copy, expp_copy, states, sample_size);
114
115 _criterion = criterion(sample_copy, sample_size, expp_copy, states);
116 _p_value = 1 - pChi(_criterion, outcomes_in_sample - 1);
117
118 delete[] expp_copy;
119 delete[] sample_copy;
120}
